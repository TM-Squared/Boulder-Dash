40  	 *ASSIGN EACH LETTER TO COMPONENTS 
41  	 */
42  	public void Caract() {
43  		int x=0;
44  		int y=0;
45  		
46  		
47  		for(int j=0; j<GameFrame.y; j++) {
48  			for(int i=0; i<GameFrame.x; i++) {
49  				if(map.getTableau()[j][i]=='M' || map.getTableau()[j][i]=='W') {
50  					Rockman.imoobjet[j][i]= new Wall(x,y);
51  				}
52  				if(map.getTableau()[j][i]=='S'|| map.getTableau()[j][i]=='5' || map.getTableau()[j][i]=='G') {
53  					Rockman.imoobjet[j][i]= new Ground(x,y);
54  				}
55  				if(map.getTableau()[j][i]=='D' || map.getTableau()[j][i]=='O') {
56  					Rockman.imoobjet[j][i]= new Diamonds(x,y);
57  				}
58  				if(map.getTableau()[j][i]=='R' || map.getTableau()[j][i]=='B' || map.getTableau()[j][i]=='*') {
59  					Rockman.imoobjet[j][i]= new Rock(x,y);
60  				}
61  				if(map.getTableau()[j][i]=='-' || map.getTableau()[j][i]==' ') {
62  					Rockman.imoobjet[j][i]= new Background(x,y);
63  				}
64  				if(map.getTableau()[j][i]=='T') {
65  					Rockman.imoobjet[j][i]= new Door(x,y);
66  				}
67  				if(map.getTableau()[j][i]=='F') {
68  					Rockman.imoobjet[j][i]= new Monsters(x,y);
69  				}
70  				x+=32;
71  			}
72  			x=0;
73  			y+=32;
74  		}
75  
76  		
77  	}

81  	 *DRAW COMPONENTS
82  	 */
83  	public void paintComponent(Graphics g) {
84  		int x=0;
85  		int y=0;
86  		//DRAW THE BACKGROUND
87  		for(int j=0; j<GameFrame.y; j++) {
88  			for(int i=0; i<GameFrame.x; i++) {
89  				g.drawImage(background.getImage(),x,y,32,32,this);
90  				x+=32;
91  			}
92  			x=0;
93  			y+=32;
94  		}
95  
96  
97  		//DRAW THE MAP
98  		for(int j=0; j<GameFrame.y; j++) {
99  			for(int i=0; i<GameFrame.x; i++) {
100 				if(Rockman.imoobjet[j][i]!=null)
101 				g.drawImage(Rockman.imoobjet[j][i].getImage(), Rockman.imoobjet[j][i].getX(), Rockman.imoobjet[j][i].getY(),32,32,this);				
102 			}
103 		}
104 			g.drawImage(KeyBoard.getRockman().getImage(),KeyBoard.getRockman().getX(),KeyBoard.getRockman().getY(),32,32, this);
105 		repaint();
106 		
107 	}
